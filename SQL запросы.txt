create table if not exists genre (
id serial primary key,
name varchar(40) not null
 );

create table if not exists performers(
id serial primary key,
name_or_pseudonym text not null
);

create table if not exists album(
id serial primary key,
name varchar(100) not null,
year_of_issue integer not null
);

create table if not exists track(
id serial primary key,
id_album integer references album(id),
name varchar(100) not null,
duration numeric(4,2) not null
);

create table if not exists genre_and_performers(
id serial primary key,
id_genre integer references genre(id),
id_performers integer references performers(id)
);

create table if not exists album_and_performers (
id serial primary key,
id_performers integer references performers(id),
id_album integer references album(id)
);

create table if not exists collection (
id serial primary key,
name varchar(120) not null unique,
year_of_issue integer not null,
id_track integer references track(id)
);
